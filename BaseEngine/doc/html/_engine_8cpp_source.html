292 bool Engine::move(int fieldNo) {
293  if (state != EngineStates::DICE_ROLLED) return false;
294  if (!getCurrentPlayerContainer().canMove() && dice.getLast() != 6) {
295  state = EngineStates::MOVE_MADE;
296  return step();
297  }
298  if (dice.getLast() == 6 && fieldNo < 0) {
299  PlayerContainer& pc = getCurrentPlayerContainer();
300  Counter* c;
301  if ((c = pc.holderPop()) != nullptr) // Jeśli w domku był pionek.
302  {
303 #ifdef _DEBUG
304  std::cout << "Popping from holder " << *c << '\n';
305 #endif
306  if (tiles[pc.getStartPos()].addToTile(c)) { // Z założenia wiele pionków może stać na polu startowym dodanie jest zawsze bezpieczne.
307  state = EngineStates::MOVE_MADE;
308  return step();
309  }
310  else pc.addToHolder(c);
311  return false;
312  }
313  }
314  else if (fieldNo < 52 && fieldNo >= 0) {
315  unsigned int distance_start = getDistance(getCurrentPlayerContainer(), fieldNo);
316  unsigned int distance = getDistance(getCurrentPlayerContainer(), fieldNo + dice.getLast());
317 #ifdef _DEBUG
318  std::cout << "Calculated distance is " << distance << " start is " << distance_start << '\n';
319 #endif
320  if (distance < 51 && distance_start < distance) {
321 #ifdef _DEBUG
322  std::cout << "Moving on board to " << fieldNo + dice.getLast() << '\n';
323 #endif
324  if (moveCounterOnBoard(fieldNo))
325  return step();
326  return false;
327  }
328  else {
329  unsigned int dest = 0;
330  if (distance > distance_start)
331  dest = distance - 51;
332  else dest = 51 - distance_start;
333 #ifdef _DEBUG
334  std::cout << "Moving to last on " << dest << '\n';
335 #endif
336  if (moveCounterToLast(fieldNo, dest))
337  return step();
338  return false;
339  }
340  }
341  else if (fieldNo > 100) {
342 #ifdef _DEBUG
343  std::cout << "Moving on last " << fieldNo << '\n';
344 #endif
345  if (moveCounterOnLast(fieldNo))
346  return step();
347  return false;
348  }
349  return false;
350 }
